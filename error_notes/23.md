class Solution(object):
    def mergeKLists(self, lists):
        """
        :type lists: List[ListNode]
        :rtype: ListNode
        """
        heads = []
        for l in lists:
            heapq.heappush(heads, l) ## l can be None! ##
        res_pre = ListNode(0)
        cur = res_pre
        while heads:
            n = heapq.heappop(heads)
            cur.next = n
            cur = cur.next
            if n.next:
                heapq.heappush(heads, n.next)
            cur.next = None
        return res_pre.next
